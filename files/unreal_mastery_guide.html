💻 C++ Programming

Basic Actor Class

// MyActor.h
#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "MyActor.generated.h"

UCLASS()
class MYGAME_API AMyActor : public AActor
{
    GENERATED_BODY()
    
public:
    AMyActor();
    
    virtual void BeginPlay() override;
    virtual void Tick(float DeltaTime) override;
    
    // Editable in Blueprint
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Stats")
    float Health = 100.0f;
    
    // Callable from Blueprint
    UFUNCTION(BlueprintCallable, Category = "Actions")
    void TakeDamage(float Amount);
    
protected:
    UPROPERTY(VisibleAnywhere)
    UStaticMeshComponent* Mesh;
};

Actor Implementation

// MyActor.cpp

#include "MyActor.h"

AMyActor::AMyActor()
{
    PrimaryActorTick.bCanEverTick = true;
    
    // Create component
    Mesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("Mesh"));
    RootComponent = Mesh;
}

void AMyActor::BeginPlay()
{
    Super::BeginPlay();
    
    // Initialize
    UE_LOG(LogTemp, Warning, TEXT("Actor spawned!"));
}

void AMyActor::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);
    
    // Per-frame logic
    AddActorLocalRotation(FRotator(0, 90 * DeltaTime, 0));
}

void AMyActor::TakeDamage(float Amount)
{
    Health -= Amount;
    
    if (Health <= 0)
    {
        Destroy();
    }
}

Input Binding (Enhanced Input)

// MyCharacter.cpp

void AMyCharacter::SetupPlayerInputComponent(
    UInputComponent* PlayerInputComponent)
{
    Super::SetupPlayerInputComponent(PlayerInputComponent);
    
    // Enhanced Input (UE5)
    if (UEnhancedInputComponent* EnhancedInput = 
        Cast<UEnhancedInputComponent>(PlayerInputComponent))
    {
        // Action bindings
        EnhancedInput->BindAction(JumpAction, 
            ETriggerEvent::Started, this, &AMyCharacter::Jump);
        
        EnhancedInput->BindAction(MoveAction,
            ETriggerEvent::Triggered, this, &AMyCharacter::Move);
    }
}

void AMyCharacter::Move(const FInputActionValue& Value)
{
    FVector2D MovementVector = Value.Get<FVector2D>();
    
    AddMovementInput(GetActorForwardVector(), MovementVector.Y);
    AddMovementInput(GetActorRightVector(), MovementVector.X);
}

UCLASS()
class MYGAME_API UMyAnimInstance : public UAnimInstance
{
    GENERATED_BODY()
    
public:
    virtual void NativeUpdateAnimation(
        float DeltaSeconds) override;
    
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly)
    float Speed;
    
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly)
    bool bIsJumping;
};

// Implementation
void UMyAnimInstance::NativeUpdateAnimation(float DeltaSeconds)
{
    Super::NativeUpdateAnimation(DeltaSeconds);
    
    ACharacter* Char = Cast<ACharacter>(TryGetPawnOwner());
    if (!Char) return;
    
    Speed = Char->GetVelocity().Size();
    bIsJumping = Char->GetMovementComponent()->IsFalling();
}

│   └── Selector (Try options)
│       ├── Sequence (Do all)
│       │   ├── Condition: CanSeePlayer?
│       │   ├── Task: MoveToPlayer
│       │   └── Task: Attack
│       │
│       └── Sequence (Patrol)
│           ├── Task: FindPatrolPoint
│           └── Task: MoveToPatrolPoint

COMPOSITE NODES:
┌─────────────────────────────────────────┐
│ Selector │ Run children until one      │
│          │ succeeds (OR logic)         │
├─────────────────────────────────────────┤
│ Sequence │ Run all children in order   │
│          │ (AND logic)                 │
├─────────────────────────────────────────┤
│ Parallel │ Run all simultaneously      │
└─────────────────────────────────────────┘

DECORATORS (Conditions):
• Blackboard-based condition
• Cooldown
• Loop
• Force success/failure

🌐 Multiplayer & Networking

Network Architecture

Client-Server Model
┌─────────────────────────────────────────────┐
│                 DEDICATED SERVER            │
│  • Authoritative                            │
│  • Runs game logic                          │
│  • Validates all actions                    │
│  • Replicates to clients                    │
└──────────────────┬──────────────────────────┘
                   │
     ┌─────────────┼─────────────┐
     │             │             │
     ▼             ▼             ▼
┌─────────┐  ┌─────────┐  ┌─────────┐
│ Client1 │  │ Client2 │  │ Client3 │
│ (Player)│  │ (Player)│  │ (Player)│
└─────────┘  └─────────┘  └─────────┘

REPLICATION:
Server → Client: Automatic
Client → Server: RPC (Remote Procedure Call)

AUTHORITY:
• Server owns most actors
• Player controller owned by client
• Replicated variables flow server→client

Variable Replication

// Replicated variable
UPROPERTY(Replicated, VisibleAnywhere)
float Health;

// In header - override GetLifetimeReplicatedProps
virtual void GetLifetimeReplicatedProps(
    TArray<FLifetimeProperty>& OutLifetimeProps) const override;

// In cpp
void AMyCharacter::GetLifetimeReplicatedProps(
    TArray<FLifetimeProperty>& OutLifetimeProps) const
{
    Super::GetLifetimeReplicatedProps(OutLifetimeProps);
    
    DOREPLIFETIME(AMyCharacter, Health);
    DOREPLIFETIME_CONDITION(AMyCharacter, Ammo, COND_OwnerOnly);
}

// RepNotify - callback when variable changes
UPROPERTY(ReplicatedUsing = OnRep_Health)
float Health;

UFUNCTION()
void OnRep_Health(); // Called on clients

Remote Procedure Calls (RPC)

// Server RPC (client calls, runs on server)
UFUNCTION(Server, Reliable)
void ServerSetHealth(float NewHealth);

// Client RPC (server calls, runs on client)
UFUNCTION(Client, Reliable)
void ClientShowDamage(float Amount);

// Multicast (runs on all clients)
UFUNCTION(NetMulticast, Reliable)
void MulticastPlayEffect(FVector Location);

// Implementation
void AMyCharacter::ServerSetHealth_Implementation(float NewHealth)
{
    Health = FMath::Clamp(NewHealth, 0, MaxHealth);
}

// Validity checks
if (GetLocalRole() == ROLE_Authority)
{
    // We're the server
    Health = NewHealth;
}
else
{
    // We're a client, tell server
    ServerSetHealth(NewHealth);
}

Rendering Optimization

Rendering Best Practices
DRAW CALL REDUCTION:
• Use Nanite for complex geometry
• Batch similar materials
• Use instanced static meshes
• LOD (Level of Detail) for all meshes

LIGHTING:
• Use Lumen for dynamic GI
• Bake static lighting when possible
• Limit dynamic lights (4-8 max)
• Use light importance volume

TEXTURES:
• Use virtual textures
• Compress appropriately
• Power-of-2 dimensions
• Mipmaps enabled

POST-PROCESSING:
• Limit screen-space effects
• Temporal AA helps with stability
• Motion blur is expensive on mobile

C++ Optimization

// Use references instead of copies
void ProcessData(const FMyStruct& Data);

// Avoid soft checks in hot paths
check(Ptr != nullptr);  // Debug only
checkSlow(Condition);   // Even slower

// Use TArray efficiently
Array.Reserve(ExpectedCount);  // Pre-allocate
Array.Empty();  // Clear, keep memory

// Object pooling
UObject* Pooled = ObjectPool.Get();
if (!Pooled) Pooled = NewObject<UMyObject>();

// Parallel execution
const bool bParallel = true;
ParallelFor(Array.Num(), [&](int32 Index)
{
    ProcessItem(Array[Index]);
}, bParallel);
