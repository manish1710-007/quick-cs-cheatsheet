Algorithm Implementations

Sorting Algorithms

Merge Sort

Time: O(n log n) | Space: O(n)
void merge(int arr[], int l, int m, int r) { int n1 = m - l + 1, n2 = r - m; int L[n1], R[n2]; for (int i = 0; i < n1; i++) L[i] = arr[l+i]; for (int j = 0; j < n2; j++) R[j] = arr[m+1+j]; int i = 0, j = 0, k = l; while (i < n1 && j < n2) arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++]; while (i < n1) arr[k++] = L[i++]; while (j < n2) arr[k++] = R[j++]; } void mergeSort(int arr[], int l, int r) { if (l < r) { int m = l + (r - l) / 2; mergeSort(arr, l, m); mergeSort(arr, m+1, r); merge(arr, l, m, r); } }

Quick Sort

Avg: O(n log n) | Worst: O(n²)
int partition(int arr[], int low, int high) { int pivot = arr[high]; int i = low - 1; for (int j = low; j < high; j++) { if (arr[j] < pivot) { i++; swap(arr[i], arr[j]); } } swap(arr[i+1], arr[high]); return i + 1; } void quickSort(int arr[], int low, int high) { if (low < high) { int pi = partition(arr, low, high); quickSort(arr, low, pi - 1); quickSort(arr, pi + 1, high); } }

Searching Algorithms

Binary Search

Time: O(log n) | Space: O(1)
int binarySearch(int arr[], int n, int target) { int low = 0, high = n - 1; while (low <= high) { int mid = low + (high - low) / 2; if (arr[mid] == target) return mid; else if (arr[mid] < target) low = mid + 1; else high = mid - 1; } return -1; // Not found } // Lower bound (first >= target) int lowerBound(int arr[], int n, int target) { int low = 0, high = n; while (low < high) { int mid = low + (high - low) / 2; if (arr[mid] < target) low = mid + 1; else high = mid; } return low; }

Time: O(n)
int linearSearch(int arr[], int n, int target) { for (int i = 0; i < n; i++) { if (arr[i] == target) return i; } return -1; }

Shortest Path Algorithms

Dijkstra's Algorithm

Time: O((V + E) log V)
void dijkstra(vector<pair<int,int>>& adj[], int V, int src) { priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq; vector<int> dist(V, INT_MAX); dist[src] = 0; pq.push({0, src}); while (!pq.empty()) { int u = pq.top().second; pq.pop(); for (auto& p : adj[u]) { int v = p.first, w = p.second; if (dist[u] + w < dist[v]) { dist[v] = dist[u] + w; pq.push({dist[v], v}); } } } }

Bellman-Ford

Time: O(V × E)
bool bellmanFord(vector<tuple<int,int,int>>& edges, int V, int src, vector<int>& dist) { dist.assign(V, INT_MAX); dist[src] = 0; for (int i = 0; i < V - 1; i++) { for (auto& e : edges) { int u = get<0>(e); int v = get<1>(e); int w = get<2>(e); if (dist[u] != INT_MAX && dist[u] + w < dist[v]) dist[v] = dist[u] + w; } } // Check for negative cycle for (auto& e : edges) { int u = get<0>(e); int v = get<1>(e); int w = get<2>(e); if (dist[u] != INT_MAX && dist[u] + w < dist[v]) return false; // Negative cycle } return true; }

0/1 Knapsack

Time: O(n × W)
int knapsack(int W, int wt[], int val[], int n) { vector<vector<int>> dp(n+1, vector<int>(W+1, 0)); for (int i = 1; i <= n; i++) { for (int w = 0; w <= W; w++) { dp[i][w] = dp[i-1][w]; // Don't take if (wt[i-1] <= w) { dp[i][w] = max(dp[i][w], val[i-1] + dp[i-1][w-wt[i-1]]); } } } return dp[n][W]; } // Space optimized: O(W) int knapsackOpt(int W, int wt[], int val[], int n) { vector<int> dp(W+1, 0); for (int i = 0; i < n; i++) { for (int w = W; w >= wt[i]; w--) { dp[w] = max(dp[w], val[i] + dp[w-wt[i]]); } } return dp[W]; }

LCS (Longest Common Subsequence)

Time: O(m × n)
int lcs(string X, string Y) { int m = X.length(), n = Y.length(); vector<vector<int>> dp(m+1, vector<int>(n+1, 0)); for (int i = 1; i <= m; i++) { for (int j = 1; j <= n; j++) { if (X[i-1] == Y[j-1]) dp[i][j] = dp[i-1][j-1] + 1; else dp[i][j] = max(dp[i-1][j], dp[i][j-1]); } } return dp[m][n]; }

LIS (Longest Increasing Subsequence)

Time: O(n log n)
int lis(vector<int>& arr) { vector<int> tail; for (int x : arr) { auto it = lower_bound(tail.begin(), tail.end(), x); if (it == tail.end()) tail.push_back(x); else *it = x; } return tail.size(); }
